018    package org.apache.commons.dbcp;
019    
020    import java.sql.SQLException;
021    
022    /**
023     * A SQLException subclass containing another Throwable
024     * 
025     * @author Dirk Verbeeck
026     * @version $Revision: 894583 $ $Date: 2009-12-30 05:48:07 -0500 (Wed, 30 Dec 2009) $
027     * @deprecated Use '(SQLException) new SQLException(msg).initCause(e)' instead; this class will be removed in DBCP 2.0
028     */
029    public class SQLNestedException extends SQLException {
030    
031        private static final long serialVersionUID = 1046151479543081202L;
032    
033        /**
034         * Constructs a new <code>SQLNestedException</code> with specified
035         * detail message and nested <code>Throwable</code>.
036         *
037         * @param msg    the error message
038         * @param cause  the exception or error that caused this exception to be
039         * thrown
040         */
041        public SQLNestedException(String msg, Throwable cause) {
042            super(msg);
043            if (cause != null){
044                initCause(cause);
045            }
046        }
047    }
